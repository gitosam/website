CREATE TABLE 'ip_location' (
	'sl_no' INT(11) NOT NULL AUTO_INCREMENT,
	'statusCode' VARCHAR(300) NULL DEFAULT NULL,
	'ipAddress' VARCHAR(300) NULL DEFAULT NULL,
	'countryCode' VARCHAR(300) NULL DEFAULT NULL,
	'countryName' VARCHAR(300) NULL DEFAULT NULL,
	'regionName' VARCHAR(300) NULL DEFAULT NULL,
	'zipCode' VARCHAR(300) NULL DEFAULT NULL,
	'latitude' VARCHAR(300) NULL DEFAULT NULL,
	'longitude' VARCHAR(300) NULL DEFAULT NULL,
	'timeZone' VARCHAR(300) NULL DEFAULT NULL,
	PRIMARY KEY ('sl_no'),
	UNIQUE INDEX 'ipAddress' ('ipAddress')
);
